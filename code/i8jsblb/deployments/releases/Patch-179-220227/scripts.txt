
  CREATE OR REPLACE FORCE VIEW "I8_MICROBANK_JS_DEV"."DEBIT_CARD_REQUESTS_VIEW" ("DEBIT_CARD_ID", "APP_USER_ID", "MOBILE_NO", "NIC", "CHANNEL_ID", "CHANNEL_NAME", "EMBOSING_NAME", "CUSTOMER_NAME", "CARD_NO", "CARD_STATUS_ID", "CARD_STATUS", "CARD_STATE_ID", "CARD_STATE", "ADDRESS", "CARD_PRODUCT_TYPE_ID", "CARD_PRODUCT_TYPE", "CARD_TYPE_CODE", "SEGMENT_ID", "SEGMENT_NAME", "CREATED_ON", "UPDATED_ON", "CREATED_BY", "UPDATED_BY", "MAILING_ADDRESS_ID", "IS_REISSUED") AS 
  SELECT D.DEBIT_CARD_ID, A.APP_USER_ID, A.MOBILE_NO, A.NIC, D.CHANNEL_ID, N.NAME CHANNEL_NAME, D.EMBOSING_NAME, A.FIRST_NAME||' '||A.LAST_NAME CUSTOMER_NAME, D.CARD_NO, S.CARD_STATUS_ID, S.NAME CARD_STATUS,
       T.CARD_STATE_ID, T.NAME CARD_STATE, R.ADDRESS, P.CARD_PRODUCT_TYPE_ID, P.DISCRIPTION CARD_PRODUCT_TYPE, P.CARD_TYPE_CODE, SG.SEGMENT_ID, SG.NAME SEGMENT_NAME, D.CREATED_ON, D.UPDATED_ON, CB.APP_USER_ID CREATED_BY, UB.APP_USER_ID UPDATED_BY, R.DEBIT_CARD_MAILING_ADDRESS_ID MAILING_ADDRESS_ID, D.RE_ISSUANCE
  FROM DEBIT_CARD D
 LEFT OUTER JOIN APP_USER A ON D.APP_USER_ID = A.APP_USER_ID
 LEFT OUTER JOIN CUSTOMER C ON A.CUSTOMER_ID = C.CUSTOMER_ID
 LEFT OUTER JOIN SEGMENT SG ON C.SEGMENT_ID = SG.SEGMENT_ID
 LEFT OUTER JOIN CARD_STATUS S ON D.CARD_STATUS_ID = S.CARD_STATUS_ID
 LEFT OUTER JOIN CARD_STATE T ON D.CARD_STATE_ID = T.CARD_STATE_ID
 LEFT OUTER JOIN SMART_MONEY_ACCOUNT M ON D.SMART_MONEY_ACCOUNT_ID = M.SMART_MONEY_ACCOUNT_ID
 LEFT OUTER JOIN CARD_PRODUCT_TYPE P ON M.CARD_PRODUCT_TYPE_ID = P.CARD_PRODUCT_TYPE_ID
 LEFT OUTER JOIN APP N ON D.CHANNEL_ID = N.APP_ID
 LEFT OUTER JOIN DEBIT_CARD_MAILING_ADDRESS R ON D.MAILING_ADDRESS_ID = R.DEBIT_CARD_MAILING_ADDRESS_ID
 LEFT OUTER JOIN APP_USER CB ON D.CREATED_BY = CB.APP_USER_ID
 LEFT OUTER JOIN APP_USER UB ON D.UPDATED_BY = UB.APP_USER_ID;
