  CREATE OR REPLACE FORCE VIEW "I8_MICROBANK_JS_PROD"."MERCHANT_DISCOUNT_CARD_VIEW" ("TRANSACTION_ID", "MOBILE_NO", "CARD_NO", "PRODUCT_NAME", "TRANSACTION_DATE", "MICROBANK_TRANSACTION_CODE", "TRANSACTION_AMOUNT", "MERCHANT_NAME", "BALANCE_AFTER_TRANSACTION", "SEGMENT_NAME", "STATUS") AS 
  SELECT T.TRANSACTION_ID, T.MOBILE_NO, T.CARD_NO, T.TRANSACTION_NAME PRODUCT_NAME, T.TRANSACTION_DATE, T.MICROBANK_TRANSACTION_CODE, 
       T.AMOUNT TRANSACTION_AMOUNT, T.MERCHANT_NAME, T.BALANCE_AFTER_TRANSACTION, S.NAME SEGMENT_NAME, T.STATUS_DESCR STATUS
  FROM TBL_TRANSACTION T
 INNER JOIN SEGMENT S ON T.SEGMENT_ID = S.SEGMENT_ID
 WHERE T.TRANSACTION_TYPE = 'C';
